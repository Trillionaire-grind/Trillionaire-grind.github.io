üîß Step 2: Backend with Express

Create server.js:

// server.js
const express = require('express');
const app = express();
const PORT = 3000;

let coffees = [];

app.use(express.static('public'));
app.use(express.json());

app.get('/api/coffees', (req, res) => {
  res.json(coffees);
});

app.post('/api/coffees', (req, res) => {
  const { name, origin, roast, notes } = req.body;
  if (name && origin) {
    coffees.push({ name, origin, roast, notes, addedAt: new Date() });
    res.status(201).json({ message: 'Coffee added!' });
  } else {
    res.status(400).json({ error: 'Name and origin are required.' });
  }
});

app.listen(PORT, () => {
  console.log(`Coffee app running at http://localhost:${PORT}`);
});

3. app.js

async function loadCoffees() {
  const res = await fetch('/api/coffees');
  const data = await res.json();
  const container = document.getElementById('coffee-list');
  container.innerHTML = '';

  data.forEach(coffee => {
    const div = document.createElement('div');
    div.className = 'coffee-item';
    div.innerHTML = `
      <strong>${coffee.name}</strong> (${coffee.origin})<br>
      Roast: ${coffee.roast || 'N/A'}<br>
      Notes: ${coffee.notes || 'None'}<br>
      Added: ${new Date(coffee.addedAt).toLocaleDateString()}
    `;
    container.appendChild(div);
  });
}

document.getElementById('coffee-form').addEventListener('submit', async (e) => {
  e.preventDefault();
  const name = document.getElementById('name').value;
  const origin = document.getElementById('origin').value;
  const roast = document.getElementById('roast').value;
  const notes = document.getElementById('notes').value;

  await fetch('/api/coffees', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ name, origin, roast, notes }),
  });

  e.target.reset();
  loadCoffees();
});

loadCoffees();
