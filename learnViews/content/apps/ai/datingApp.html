Create a server.js file in your root directory:

// server.js
const express = require('express');
const app = express();
const PORT = 3000;

let users = [];
let likes = {};

app.use(express.static('public'));
app.use(express.json());

// Get all users
app.get('/api/users', (req, res) => {
  res.json(users);
});

// Add a new user
app.post('/api/users', (req, res) => {
  const { name, bio } = req.body;
  if (name && bio) {
    const id = users.length + 1;
    users.push({ id, name, bio });
    res.status(201).json({ message: 'User added' });
  } else {
    res.status(400).json({ error: 'Name and bio required' });
  }
});

// Like a user
app.post('/api/like', (req, res) => {
  const { fromId, toId } = req.body;
  if (!likes[fromId]) likes[fromId] = [];
  likes[fromId].push(toId);
  res.json({ message: 'Like recorded' });
});

app.listen(PORT, () => {
  console.log(`Server running at http://localhost:${PORT}`);
});

async function loadUsers() {
  const res = await fetch('/api/users');
  const users = await res.json();

  const container = document.getElementById('user-list');
  container.innerHTML = '';

  users.forEach(user => {
    const div = document.createElement('div');
    div.className = 'user';
    div.innerHTML = `<strong>${user.name}</strong><br>${user.bio}<br>
      <button onclick="likeUser(${user.id})">Like</button>`;
    container.appendChild(div);
  });
}

document.getElementById('new-user-form').addEventListener('submit', async (e) => {
  e.preventDefault();
  const name = document.getElementById('name').value;
  const bio = document.getElementById('bio').value;

  await fetch('/api/users', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ name, bio }),
  });

  loadUsers();
});

async function likeUser(toId) {
  await fetch('/api/like', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ fromId: 1, toId }), // hardcoded user for demo
  });
  alert('You liked this user!');
}

loadUsers();
